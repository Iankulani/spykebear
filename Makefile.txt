# SPYKE BEAR Makefile

.PHONY: help install uninstall run docker-build docker-run docker-stop clean test lint

# Colors
GREEN := \033[0;32m
YELLOW := \033[1;33m
RED := \033[0;31m
NC := \033[0m

help:
	@echo "$(GREEN)SPYKE BEAR - Available Commands$(NC)"
	@echo ""
	@echo "$(YELLOW)Installation:$(NC)"
	@echo "  make install          - Install dependencies and setup"
	@echo "  make install-linux    - Linux installation"
	@echo "  make install-macos    - macOS installation"
	@echo "  make install-windows  - Windows installation"
	@echo ""
	@echo "$(YELLOW)Development:$(NC)"
	@echo "  make run              - Run Spyke Bear"
	@echo "  make dev              - Run with debug mode"
	@echo "  make test             - Run tests"
	@echo "  make lint             - Run linters"
	@echo ""
	@echo "$(YELLOW)Docker:$(NC)"
	@echo "  make docker-build     - Build Docker image"
	@echo "  make docker-run       - Run Docker container"
	@echo "  make docker-stop      - Stop Docker container"
	@echo "  make docker-clean     - Remove Docker containers"
	@echo ""
	@echo "$(YELLOW)Utilities:$(NC)"
	@echo "  make clean            - Clean temporary files"
	@echo "  make uninstall        - Uninstall Spyke Bear"
	@echo "  make update           - Update dependencies"

install:
	@echo "$(GREEN)[*] Installing Spyke Bear...$(NC)"
	@chmod +x install.sh
	@./install.sh

install-linux:
	@echo "$(GREEN)[*] Installing for Linux...$(NC)"
	@bash install.sh

install-macos:
	@echo "$(GREEN)[*] Installing for macOS...$(NC)"
	@bash install.sh

install-windows:
	@echo "$(GREEN)[*] Installing for Windows...$(NC)"
	@powershell -ExecutionPolicy Bypass -File install.ps1

run:
	@echo "$(GREEN)[*] Starting Spyke Bear...$(NC)"
	@python3 spykebear.py

dev:
	@echo "$(GREEN)[*] Starting in development mode...$(NC)"
	@export FLASK_ENV=development && python3 spykebear.py --debug

test:
	@echo "$(GREEN)[*] Running tests...$(NC)"
	@pytest tests/ -v --cov=. --cov-report=term-missing

lint:
	@echo "$(GREEN)[*] Running linters...$(NC)"
	@black --check spykebear.py
	@flake8 spykebear.py --max-line-length=120
	@pylint spykebear.py --exit-zero

docker-build:
	@echo "$(GREEN)[*] Building Docker image...$(NC)"
	@docker build -t spykebear:latest .

docker-build-alpine:
	@echo "$(GREEN)[*] Building Alpine Docker image...$(NC)"
	@docker build -f Dockerfile.alpine -t spykebear:alpine .

docker-run:
	@echo "$(GREEN)[*] Running Docker container...$(NC)"
	@docker-compose up -d

docker-run-alpine:
	@echo "$(GREEN)[*] Running Alpine Docker container...$(NC)"
	@docker-compose -f docker-compose.alpine.yml up -d

docker-stop:
	@echo "$(GREEN)[*] Stopping Docker container...$(NC)"
	@docker-compose down

docker-clean:
	@echo "$(GREEN)[*] Cleaning Docker containers...$(NC)"
	@docker-compose down -v
	@docker rmi spykebear:latest 2>/dev/null || true

clean:
	@echo "$(GREEN)[*] Cleaning temporary files...$(NC)"
	@rm -rf __pycache__/
	@rm -rf .pytest_cache/
	@rm -rf .coverage
	@rm -rf htmlcov/
	@rm -rf dist/
	@rm -rf build/
	@rm -rf *.egg-info/
	@find . -type d -name "__pycache__" -exec rm -rf {} + 2>/dev/null || true
	@find . -type f -name "*.pyc" -delete
	@echo "$(GREEN)[✓] Clean complete$(NC)"

uninstall:
	@echo "$(YELLOW)[!] This will remove Spyke Bear and all data. Continue? (y/N)$(NC)"
	@read -r confirm; if [ "$$confirm" != "y" ] && [ "$$confirm" != "Y" ]; then exit 1; fi
	@echo "$(RED)[*] Removing Spyke Bear...$(NC)"
	@rm -rf ~/.spykebear
	@rm -f /usr/local/bin/spykebear
	@rm -f ~/.local/share/applications/spykebear.desktop
	@echo "$(GREEN)[✓] Uninstall complete$(NC)"

update:
	@echo "$(GREEN)[*] Updating dependencies...$(NC)"
	@pip install --upgrade -r requirements.txt
	@echo "$(GREEN)[✓] Update complete$(NC)"
